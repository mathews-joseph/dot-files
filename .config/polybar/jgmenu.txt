# checkout-menu
Lock,^checkout(lock)
Logout,^checkout(logout)
Reboot,^checkout(reboot)
Shutdown,^checkout(shutdown)

# lock sub-menu
^tag(lock)
Sure?,i3lock-fancy -p

# logout sub-menu
^tag(logout)
Sure?,i3-msg exit

# reboot sub-menu
^tag(reboot)
Sure?,systemctl -i reboot

# shutdown sub-menu
^tag(shutdown)
Sure?,systemctl -i poweroff
